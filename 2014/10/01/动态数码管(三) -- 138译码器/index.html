动态数码管(三) -- 138译码器

在普中单片机上结合电路图可以看到138译码器有ABC3个输入接口,输出排线(J15)在数码管位选(J16)左边.因为G1已经接VCC,G2A,G2B已接地,所以不用在进行控制,可以直接使用.

我们分别用三根引线将ABC分别接在P2.2->2.4.用短路帽将J15和J16相连,那么138译码器就和动态数码管连接完毕了.

输入端ABC组成的3位2进制可表示0~7八个数,表示相应的Y,匹配8位数码管,输出为低电平.

例如:ABC输入为010即2(dec),输出为Y2=0,其他为均为1,此时低位第3位数码管亮.

例程:

#include<reg51.h>  
#define dula P0  
#define uint unsigned int   
#define uchar unsigned char  
/*定义译码器输入口*/   
sbit LSA=P2^2;  
sbit LSB=P2^3;  
sbit LSC=P2^4;  
/*位选数码表*/  
uchar code wetab[]={  
0xfe,0xfd,0xfb,0xf7,0xef,0xdf,0xbf,0x7f};  
/*段选数码表*/  
uchar code dutab[]={  
0x3f,0x06,0x5b,0x4f,0x66,0x6d,0x7d,0x07,  
0x7f,0x6f,0x77,0x7c,0x39,0x5e,0x79,0x71  
};  
void delay(uint z);
void main()  
{    
    uchar i,j;  
    for(i=0;i<8;i++){   
        switch(i){  
            case(0):  
                LSA=0;LSB=0;LSC=0; break;  //000,即为低位第1盏灯亮,满足条件时跳出循环,准备发送段码
            case(1):  
                LSA=1;LSB=0;LSC=0; break;   
            case(2):  
                LSA=0;LSB=1;LSC=0; break;  
            case(3):  
                LSA=1;LSB=1;LSC=0; break; 
            case(4):
                LSA=0;LSB=0;LSC=1; break;  
            case(5):  
                LSA=1;LSB=0;LSC=1; break;
            case(6):  
                LSA=0;LSB=1;LSC=1; break;  
            case(7):  
                LSA=1;LSB=1;LSC=1; break;      
        }
    dula=dutab[i];//发送段选   
    delay(100);    //扫描间隔时间设定  
    while(j--);      
    dula=0;    //消隐
    }//回到for循环  
}  
void delay(uint z)
{  
uint x,y;  
for(x=z;x>0;x--)  
for(y=110;y>0;y--);   
}
