红外线运用扩展之遥控显示数码　　

  • 通过时间来判断0和,为了保证时间的准确性,我们用定时器来产生这个时间
  • 因为无论是引导码,0码还是1码,都是由1个高电平和1个低电平组成,且都是由1个下降沿开启,根据这个特点,我们将这个信号接到单片机的P3.2口,即中断0引脚,利用中断0对这1帧的信号进行接收,这样就不会错过这个信号.
#include<reg51.h>
#define uint unsigned int 
#define uchar unsigned char 
#define dula P0
#define wela P1                          

/*位选数码表*/
uchar code wetab[]={  
0xfe,0xfd,0xfb,0xf7,0xef,0xdf,0xbf,0x7f};  
/*段选数码表,可自定义*/  
uchar code dutab[]={  
0x3f,0x06,0x5b,0x4f,
    0x66,0x6d,0x7d,0x07,
    0x7f,0x6f,0x77,0x7c,
    0x39,0x5e,0x79,0x71};  

sbit ir=P3^2;    //红外输出引脚与单片机P3.2口相连

uchar time;
uchar irdata[33];//用于存放接收到的元素,一共1位引导+8*4(客户码1+客户码2+操作码+操作码反码)
uchar bitnum;//标记位,用来表示接收到的数据是第几个元素
uchar startflag; //标记位
uchar irok;          //标记一帧数据收完成
uchar ircode[4];    //用来存放非引导码的4组数据
uchar irprosok;    //红外处理完毕的标志位  
uchar disnum[8]; //数码管显示用,存放拆成8位的ircode

void int0Init();

delay(uchar x)
{    
    while(x--);
}


void Timer0Init()     //定时器0初始化,每产生一次中断经历的时间为256*1us=0.256ms
{
  TMOD=0x02;//定时器0工作方式2,TH0是重装值,TL0是初值
  TH0=0x00; //重载值
  TL0=0x00; //初始化值
  ET0=1;//开中断
  TR0=1;    //启动定时器
}

 void irpros()//红外处理,将每8个2进制数转换成2个16进制数
 {
 /*0码用时0.56*2=1.12ms,用了1.15/0.256=4.375个机器周期
 1码用时0.56+0.18=2.245ms,用了2.245/0.256=8.76个机器周期
 所以我们取一个中间值6,小于6为0,大于6为1*/
     uchar mun,k,i,j;
    k=1;//第1个数为引导码,所以从第二位开始读取
    for(j=0;j<4;j++){
        for(i=0;i<8;i++)//客户码&操作码共4组数据
        {
            mun=mun>>1;//进位下面进行了进位,所以需要右移
        /*因为time的值已经赋给irdata数组中,    所以用irdata代替time判断时间*/
             if(irdata[k]>6)    
            {
                //收到的数据为逆序
                mun=mun|0x80;//最高位进1    
            }
            k++;
        }
        ircode[j]=mun; }
        irprosok=1;    //说明红外处理完毕
 }

void irwork()     //对红外码值进行转换
{
    disnum[0]=ircode[0]/16;
    disnum[1]=ircode[0]%16;
    disnum[2]=ircode[1]/16;
    disnum[3]=ircode[1]%16;
    disnum[4]=ircode[2]/16;
    disnum[5]=ircode[2]%16;
    disnum[6]=ircode[3]/16;
    disnum[7]=ircode[3]%16;
}

void display()
{
    uchar i;
    for(i=0;i<6;i++)
    {
        P0=0xff;
        wela=1;
        wela=0;
        P0=dutab[disnum[i]];
        dula=1;
        dula=0;
        P0=wetab[i];
        wela=1;
        wela=0;
        delay(300);
    }
}

void main()
{
    Timer0Init();
    int0inInt();
    while(1)
    {
        if(irok==1)     //表示一帧数据接受完成
        {
            irpros(); //红外处理
            irok=0;     //清零
        }
        if(irprosok==1)      //处理完毕
        {
            irwork();     //转换成数码管数值
            irprosok=0;      //清零
        }
        display();     //显示
    }

}

void int0Init()//外部中断0初始化
{
    IT0=1;    //设置为下降沿触发方式
    EX0=1;    //打开外部中断
    EA=1;    //打开总中断
}

void int0() interrupt 0       //利用外部中断0接收每一帧的数据
{
    /*计算接收到的每个数字的时间来辨别是引导码还是0,1
    当进入中断时,startflag未赋值,进入else语句,将time清零,以便计算,
    startflag赋1,进入if进行数据处理*/
    if(startflag)     //标记
    {      
         /*接收到的引导码为9ms低电平加4.5ms高电平=13.5ms
        13.5/0.256=52.73个机器周期,考虑到外界环境干扰,
        引导码可能会有一些变形,
        所以我们在程序上通常会给出一定的取值范围*/
        if(time>32 && time<63)//8~16ms内,我们就把它认为是引导码
        {
            bitnum=0;//检测到引导码后,将标记位清零
        }
        irdata[bitnum]=time;    //装入数据
        time=0;       //清零,因为下个数据还需用到time
        bitnum++;
        if(bitnum=33) //自动加1,至满时清零
        {
            bitnum=0;
            irok=1;        //表示这一帧数据接受完成
        }
    }
    else{
        time=0;
        startflag=1;
    }
}

void Timer0() interrupt 1    //定时器0中断入口为1
{
     time++;            //定时器每中断一次+1
}

单片机学习笔记之红外线

基本原理

将基带二进制信号调制为脉冲串信号,通过红外发射管发射.

NEC协议

引导码+用户码(8位)+用户码(8位)+键数据码+键数据码反码

  1. 引导码:先发送9ms的AGC（自动增益控制）高脉冲,接着发送4.5ms的起始低电平

  2. 如果你一直按某个按键，一串信息也只能发送一次，一直按着，发送的则是以110ms为周期的重复码
    重复码的格式是由9ms的AGC高电平和4.5ms的低电平及一个560us的高电平组成

    • 逻辑1由560us的高电平和1.69ms的低电平脉冲表示
    • 逻辑0的是由560us的高电平和565us的低电平脉冲表示

  3. 接收到的信号和发送的信号相反即低位在前高位在后,需经过排序
    -所以一开始接收到9ms的电平为低电平,也就是说脉冲为负跳变,为进入外部中断提供条件

产生下降沿,进入外部中断0的中断函数,延时一下,检测IO口是否还是低电平,是就等待9ms的低电平过去。
等待完9ms低电平过去，再去等待4.5ms的高电平过去.
开始接收传送的4组数据
先等待560us的低电平过去
检测高电平的持续时间，如果超过1.12ms那么是高电平( 高电平的的持续时间为1.69ms，低电平的持续时间为565us)
检测接收到的数据和数据的反码进行比较，是否等到的数据是一样的。

例程

/*******************************************************************************
* 实验名               : 1602显示红外线值实验
* 使用的IO     : 电机用P1口,键盘使用P3.0、P3.1、P3.2、P3.3
* 实验效果   : LCD1602显示出读取到的红外线的值
*******************************************************************************/ 
#include<reg51.h>
#define uint unsigned int 
#define uchar unsigned char  
#include"lcd.h"

sbit IRD=P3^2;

uchar code CDIS1[13]={" Red Control "};
uchar code CDIS2[13]={" IR-CODE:--H "};
uchar IrValue[6];
uchar Time;
void IrInit();
void Delay(uint);

void main()
{
    uchar i;
    IrInit();    
    LcdInit();
    LcdWriteCom(0x80);
    for(i=0;i<13;i++)
    {
        LcdWriteData(CDIS1[i]);    
    }
    LcdWriteCom(0x80+0x40);
    for(i=0;i<13;i++)
    {
        LcdWriteData(CDIS2[i]);    
    }
    while(1)
    {
        IrValue[5]=IrValue[2]>>4;              //高位
        IrValue[6]=IrValue[2]&0x0f;        //低位    
        if(IrValue[5]>9)
        {
            LcdWriteCom(0xc0+0x09);            //设置显示位置
            LcdWriteData(0x37+IrValue[5]);    //将数值转换为该显示的ASCII码
        }
        else
        {
            LcdWriteCom(0xc0+0x09);
            LcdWriteData(IrValue[5]+0x30);    //将数值转换为该显示的ASCII码
        }    
        if(IrValue[6]>9)
        {
            LcdWriteCom(0xc0+0x0a);
            LcdWriteData(IrValue[6]+0x37);        //将数值转换为该显示的ASCII码
        }
        else
        {
            LcdWriteCom(0xc0+0x0a);
            LcdWriteData(IrValue[6]+0x30);        //将数值转换为该显示的ASCII码
        }    
    }
}

* 函数名 : Delay()
* 函数功能           : 延时
*******************************************************************************/

void Delay(unsigned int x)   //0.14ms误差 0us
{
 unsigned char i;
  while(x--)
 {
  for (i = 0; i<13; i++)
 {}
 }
}
/*******************************************************************************
* 函数名 : IrInit()
* 函数功能           : 初始化红外线接收
*******************************************************************************/

void IrInit()
{
    IT0=1;//下降沿触发
    EX0=1;//打开中断0允许
    EA=1;    //打开总中断

    IRD=1;//初始化端口
}
/*******************************************************************************
* 函数名 : ReadIr()
* 函数功能           : 读取红外数值的中断函数
*******************************************************************************/

void ReadIr() interrupt 0    //跳线连接P3.2(外部中断0)
{
    uchar j,k;
    uint err;
    Time=0;                     
    Delay(70);        //进入中断后先进行延时

    if(IRD==0)        //确认是否真的接收到正确的脉冲
    {     

        err=1000;                //1000*10us=10ms,超过说明接收到错误的信号
        /*当两个条件都为真是循环，如果有一个条件为假的时候跳出循环，免得程序出错的时
        侯，程序死在这里*/    
        while((IRD==0)&&(err>0))    //等待前面9ms的低电平过去          
        {            
            Delay(1);
            err--;
        } 
        if(IRD==1)            //如果正确等到9ms低电平
        {
            err=500;
            while((IRD==1)&&(err>0))         //等待4.5ms的起始高电平过去
            {
                Delay(1);
                err--;
            }
            for(k=0;k<4;k++)        //共有4组数据
            {                
                for(j=0;j<8;j++)    //接收一组数据
                {

                    err=60;        
                    while((IRD==0)&&(err>0))//等待信号前面的560us低电平过去
//                    while (!IRD)
                    {
                        Delay(1);
                        err--;
                    }
                    err=500;
                    while((IRD==1)&&(err>0))     //计算高电平的时间长度。
                    {
                        Delay(1);//0.14ms
                        Time++;
                        err--;
                        if(Time>30)
                        {
                            EX0=1;
                            return;
                        }
                    }
                    IrValue[k]>>=1;     //k表示第几组数据
                    if(Time>=8)            //如果高电平出现大于565us，那么是1
                    {
                        IrValue[k]|=0x80;
                    }
                    Time=0;        //用完时间要重新赋值                            
                }
            }
        }
        if(IrValue[2]!=~IrValue[3])
        {
            return;
        }
    }            
}

单片机学习笔记之温度显示

DS18B20简介

  1. 仅需要一条口线即可实现微处理器与DS18B20 的双向通讯
  2. 测量范围为 -55 ℃ ~+ 125 ℃ ; 在 -10~+ 85°C 范围内,精度为 ± 0.5°C
  3. 多个 DS18B20 可以并联在惟一的单线上
  4. 分辨率可通过程序设定 9~12 位

单线总线特点

  • 单总线即只有一根数据线,系统中的数据交换,控制都由这根线完成。
  • 单总线通常要求外接一个约为 4.7K—10K 的上拉电阻，这样，当总线闲置时其状态为高电平

DS18B20工作方式

  • 初始化

    主机首先发出一个480－960微秒的低电平脉冲，然后释放总线变为高电平，并在随后的480微秒时间内对总线进行检测，如果有低电平出现说明总线上有器件已做出应答。若无低电平出现一直都是高电平说明总线上无器件应答(初始化不成功)。
    做为从器件的DS18B20在一上电后就一直在检测总线上是否有480－960微秒的低电平出现，如果有，在总线转为高电平后等待15－60微秒后将总线电平拉低60－240微秒做出响应存在脉冲，告诉主机本器件已做好准备。若没有检测到就一直在检测等待。

    uchar Ds18b20Init()
    {

      uint i;
  DSPORT=0;             //将总线拉低480us~960us
  i=70;    
  while(i--);          //延时642us
  DSPORT=1;            //然后拉高总线，如果DS18B20做出反应会将在15us~60us后总线拉低
  i=0;
  while(DSPORT)    //等待DS18B20拉低总线
  {
      i++;
      if(i>5000)//等待>5MS
          return 0;//初始化失败    
  }
  return 1;//初始化成功

    }

  • 数据发送
    写周期最少为60微秒，最长不超过120微秒。写周期一开始做为主机先把总线拉低1微秒表示写周期开始。随后若主机想写0，则将总线置为低电平，若主机想写1，则将总线置为高电平，持续时间最少60微秒直至写周期结束，然后释放总线为高电平至少1微秒给总线恢复 。而DS18B20则在检测到总线被拉底后等待15微秒然后从15us到45us开始对总线采样，在采样期内总线为高电平则为1，若采样期内总线为低电平则为0。

    void Ds18b20WriteByte(uchar dat)
    {

      uint i,j;
  for(j=0;j<8;j++)
  {
      DSIO=0;            //每写入一位数据之前先把总线拉低1us
      i++;
      DSIO=dat&0x01; //将数据写入总线,从最后一位开始读取
      i=6;
      while(i--); //延时68us，持续时间最少60us
      DSIO=1;    //然后释放总线，至少1us给总线恢复时间才能接着写入第二个数值
      dat>>=1;
  }

    }

  • 数据发送
    对于读数据操作时序也分为读0时序和读1时序两个过程。
    读周期是从主机把单总线拉低1微秒之后就得释放单总线为高电平，以让DS18B20把数据传输到单总线上。作为从机DS18B20在检测到总线被拉低1微秒后，便开始送出数据，若是要送出0就把总线拉为低电平直到读周期结束。若要送出1则释放总线为高电平。主机在一开始拉低总线1微秒后释放总线，然后在包括前面的拉低总线电平1微秒在内的15微秒时间内完成对总线进行采样检测，采样期内总线为低电平则确认为0。采样期内总线为高电平则确认为1。完成一个读时序过程，至少需要60微秒才能完成

    uchar Ds18b20ReadByte()
    {

      uchar byte,bi;
  uint i,j;    
  for(j=8;j>0;j--)
  {
      DSIO=0;//先将总线拉低1us
      i++;
      DSIO=1;//然后释放总线
      i++;
      i++;//延时6us等待数据稳定
      bi=DSIO;     //读取数据，从最低位开始读取
      byte=(byte>>1)|(bi<<7); /*将byte左移一位，然后与上右移7位后的bi，注意移动之后移掉那位补0。*/

      i=4;        //读取完之后等待48us再接着读取下一个数
      while(i--);
  }                
  return byte;

    }

操作步骤

初始化→ROM操作指令→存储器操作指令→数据传输。
其工作时序包括：

  • 初始化时序
  • 写时序
  • 读时序

简单的读取温度值的步骤如下：

  1. 跳过ROM操作
  2. 发送温度转换命令
  3. 跳过ROM操作
  4. 发送读取温度命令

  5. 读取温度值

    int Ds18b20ReadTemp()
    {

     unsigned int temp=0;
 unsigned char tmh,tml;
 Ds18b20ChangTemp();    //先写入转换命令
 Ds18b20ReadTempCom();//然后等待转换完后发送读取温度命令

 tmh=Ds18b20ReadByte();    //再读高字节
 temp=tmh;
 temp<<=8;
 temp|=tml;
 return temp;

    }

单片机学习笔记之中断

51单片机的中断有5个(具有2个中断优先级,可实现2级中断服务和程序嵌套)

-外部中断*2    (INT0/1 非)  
-定时/计数器中断*2    (IT0/1) 
-串口发送/接受中断*1    (TX/RX)  

||       引 脚       ||        第二功能详解         ||
||   P3.0/RxD     ||        串行输入口              ||
||   P3.1/TxD     ||        串行输出口             ||
||   P3.2/INT0     ||        外部中断0输入           ||
||   P3.3/INT1     ||        外部中断1输入           ||
||   P3.4/T0     ||    定时器0,外部计数器输入    ||
||   P3.5/T1     ||    定时器1,外部计数器输入    ||
||   P3.6/WR     ||        外部写控制               ||
||   P3.7/RD       ||        外部读控制            ||

+80C51中断系统结构

中断—>中断标志(TCON&SCON)—>中断总允许(EA)&源允许(IE)—>中断优先级(IP)

1. （P3.2）可由IT0(TCON.0)选择其为低电平有效还是下降沿有效。当CPU检测到P3.2引脚上出现有效的中断信号时，中断标志IE0(TCON.1)置1，向CPU申请中断。
2. (P3.3）可由IT1(TCON.2)选择其为低电平有效还是下降沿有效。当CPU检测到P3.3引脚上出现有效的中断信号时，中断标志IE1(TCON.3)置1,向CPU申请中断。
3. TF0（TCON.5），片内定时/计数器T0溢出中断请求标志。当定时/计数器T0发生溢出时，置位TF0，并向CPU申请中断。
4. TF1（TCON.7），片内定时/计数器T1溢出中断请求标志。当定时/计数器T1发生溢出时，置位TF1，并向CPU申请中断。
5. RI（SCON.0）或TI（SCON.1），串行口中断请求标志。当串行口接收完一帧串行数据时置位RI或当串行口发送完一帧串行数据时置位TI，向CPU申请中断。

+中断请求标志

-意义:告诉CPU有那个中断源发出了中断请求,让CPU选择是否响应.

1.TCON的中断标志

TCON

    位              7        6        5        4        3        2        1        0
字节地址:88H    TF1         TR1      TF0      TR0      IE1      IT1      IE0      IT0
  • IT0（TCON.0），外部中断0触发方式控制位。
    • 当IT0=0时，低电平触发
    • 当IT0=1时，负跳变沿触发
  • IE0/IE1（TCON.1/TCON.3），外部中断0/1中断请求标志位。
  • IT1（TCON.2），外部中断1触发方式控制位。
  • TF0/TF1（TCON.5/TCON.7），定时/计数器T0/T1溢出中断请求标志位。

2.SCON的中断标志

SCON

    位              7    6    5    4    3    2    1    0
字节地址:88H                                 TI  RI
  • RI（SCON.0），串行口接收中断标志位。
    • 当允许串行口接收数据时，每接收完一个串行帧，由硬件置位RI。RI必须由软件清除。
  • TI（SCON.1），串行口发送中断标志位。
    • 当CPU将一个发送数据写入串行口发送缓冲器时，就启动了发送过程。每发送完一个串行帧，由硬件置位TI。CPU响应中断时，不能自动清除TI，TI必须由软件清除。

3. 中断允许寄存器IE

位        D7     D6     D5        D4        D3        D2        D1        D0
功能        EA                     ES        ET1       EX1        ET0       EX0
  • EA:总中断允许位。
    =0：关闭中断
    =1：CPU开放所有中断
  • /*ES：串行口中断允许位。
    =0：禁止串行中断。
    =1：允许串行中断。
  • ET1/ET0：定时器1/0中断允许位
    =0时：禁止相应的定时器中断
    =1时：允许相应的定时器中断
  • EX1和EX0：分别是外部中断1和外部中断0的中断允许位
    =0：禁止相应的外部中断
    =1：允许相应的外部中断*/

4. 中断优先寄存器IP

位        D7     D6     D5        D4        D3        D2        D1        D0
功能        EA                     PS        PT1       PX1        PT0       PX0

一般都默认低优先级,不需要特别更改

中断优先级

中断响应条件

-中断源有中断请求  
-此中断源的中断允许为1  
-CPU开中断(即EA=1)

中断函数写法

void 函数名 () interrupt 中断编号 
{
    ;
}

开启外部中断的步骤:

1. 设置外部中断触发方式(IT0或IT1)  
2. 打开相应的外部中断允许(EX0或EX1)  
3. 打开总中断(即EA=1)

例程:

#include<reg51.h>
#include<intrins.h>  
#define uint unsigned int 
#define uchar unsigned char  
#define LED P2 
sbit K3=P3^2;  
sbit K4=P3^3;  

void IntConfig();
void delay(z);
uchar KEY=0;

void main()  
{  
    P2=0xfe;
    IntConfig(); 
    while(1)
    {
        if(KEY)
            LED=_crol_(LED,1);
        else
            LED=_cror_(LED,1);
        delay(1000);
    }


}  

void IntConfig()//设置外部中断
{
    /*设置INT0*/  
    IT0=1;//设置外部中断0为负翘变沿触发方式  
    EX0=1;//打开INT0的中断允许  
    /*设置INT1*/  
    IT1=1;//设置外部中断1为负翘变沿触发方式  
    EX1=1;//打开INT1的中断允许   
    EA=1;  
}

void Int0()    interrupt 0        //外部中断0的中断函数
{
    delay(1);     //延时消抖
    if(K3==0)
        KEY=1;
}


void Int1() interrupt 2        //外部中断1的中断函数
{
    delay(1);        //延时消抖
    if(K4==0)
        KEY=0;    
}  

void delay(uint z)
{
    uint x,y;
    for(x=z;x>0;x--)
        for(y=110;y>0;y--);    
}
